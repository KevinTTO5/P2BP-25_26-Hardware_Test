# laptop/deepstream/config_infer_primary.txt
#
# Gst-nvinfer configuration for the DS 9.0 MV3DT laptop pipeline.
#
# Detector: PeopleNet — the ONLY detector wired into this pipeline. This
# matches NVIDIA's DS 9.0 MV3DT reference documentation and is intentionally
# enforced by this plan. Notion page 337b5d58-7212-81e1-b07a-d510d9605bbb
# §9.3 Option A.
#
# Precision: FP16 by default. Per the DS 9.0 breaking-changes table in the
# deepstream-9-docs skill (reference.md, "DS 9.0 Breaking Changes"), INT8
# calibration for TAO models was removed in DS 9.0. We only flip to INT8 if
# the pinned PeopleNet model version (§9.3) ships an INT8 calibration cache.
#
# Field names verified via:
#   - Context7 /websites/nvidia_metropolis_deepstream_dev-guide
#   - WebFetch https://docs.nvidia.com/metropolis/deepstream/dev-guide/text/DS_plugin_gst-nvinfer.html
#
# ----------------------------------------------------------------------------
# Future-work alternative detector (NOT installed, NOT configured here):
#   yolo11n (ultralytics yolo11n.pt) — the only approved alternative per the
#   plan in .cursor/plans/laptop_ds9_scripted_testing_*.plan.md. See the
#   deepstream-9-docs skill entry for marcoslucianops/DeepStream-Yolo
#   (reference.md §"Third-Party / Community") when wiring it in a future
#   iteration. This file deliberately contains no YOLO block, commented or
#   otherwise.
# ----------------------------------------------------------------------------

[property]
gpu-id=0

# Pixel normalization per the NGC PeopleNet model card.
net-scale-factor=0.0039215697906911373
model-color-format=0

# Model artifacts (written by laptop/scripts/25_prepare_models.sh into
# laptop/deepstream/models/peoplenet/). The exact filenames come from the
# PeopleNet NGC artifact and are stable across the deployable_quantized_v2.6.x
# family; 25_prepare_models.sh fails fast if either path is missing.
tlt-encoded-model=models/peoplenet/resnet34_peoplenet_int8.etlt
tlt-model-key=tlt_encode
labelfile-path=models/peoplenet/labels.txt

# Engine built on first run. PeopleNet ships an INT8 calibration cache
# alongside the .etlt; if you want to exercise the DS 9.0 FP16 default
# instead, comment out int8-calib-file and set network-mode=2 below.
int8-calib-file=models/peoplenet/resnet34_peoplenet_int8.txt
model-engine-file=models/peoplenet/resnet34_peoplenet_int8.etlt_b8_gpu0_int8.engine

input-dims=3;544;960;0
uff-input-blob-name=input_1
output-blob-names=output_cov/Sigmoid;output_bbox/BiasAdd

batch-size=8

# 0=FP32, 1=INT8, 2=FP16. DS 9.0 default is FP16 for TAO detectors because
# INT8 calibration was removed for TAO in 9.0; PeopleNet's bundled INT8
# cache is still honoured when present, so we keep network-mode=1 here and
# let Gst-nvinfer fall back cleanly if the cache is absent.
network-mode=1

num-detected-classes=3
interval=0
gie-unique-id=1
process-mode=1
network-type=0
cluster-mode=2

maintain-aspect-ratio=1

[class-attrs-all]
pre-cluster-threshold=0.2
nms-iou-threshold=0.5
topk=20
